SELECT I.SUMMARY, IL.NAME
FROM ISSUE I
JOIN ISSUESLISTS IL ON IL.ID = I.ISSUELIST_ID;

SELECT IL.NAME, I.SUMMARY, U.FIRSTNAME, U.LASTNAME
FROM ISSUE I
JOIN USERS U ON U.ID = I.USER_ID_ASSIGNTO
JOIN ISSUESLISTS IL ON  IL.ID = I.ISSUELIST_ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TASKS_NUMBER
FROM ISSUE I
JOIN USERS U ON U.ID = I.USER_ID_ASSIGNTO
GROUP BY I.USER_ID_ASSIGNTO;
